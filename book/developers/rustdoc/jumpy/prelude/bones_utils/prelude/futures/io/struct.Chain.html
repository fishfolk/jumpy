pub fn get_ref(&self) -> (&R1, &R2)

Gets references to the underlying readers.

Examples
use futures_lite::io::{AsyncReadExt, Cursor};

let r1 = Cursor::new(b"hello");
let r2 = Cursor::new(b"world");

let reader = r1.chain(r2);
